\section{Ordenaçăo Interna}
\begin{frame}
\frametitle{Introduçăo - Conceitos Básicos}
\begin{itemize}
\item Ordenar: processo de \textbf{rearranjar} um conjunto de objetos em uma
ordem ascendente ou descendente.
\item A ordenaçăo visa facilitar a \textbf{recuperaçăo} posterior de itens do conjunto
ordenado.
\begin{itemize}
\item Dificuldade de se utilizar um catálogo telefônico se os nomes das
pessoas năo estivessem listados em ordem alfabética.
\end{itemize}
\item Notaçăo utilizada nos algoritmos:
\begin{itemize}
\item Os algoritmos trabalham sobre os \textbf{registros} de um arquivo.
\item Cada registro possui uma \textbf{chave} utilizada para controlar a
ordenaçăo.
\item Podem existir \textbf{outros componentes} em um registro
\end{itemize}

\end{itemize}
\end{frame}




\begin{frame}
\frametitle{Introduçăo - Conceitos Básicos} \footnotesize
\begin{block}{Estrutura de um registro}

\begin{itemize}
\item Qualquer tipo de chave sobre o qual exista uma \textbf{regra de ordenaçăo
bem-definida} pode ser utilizado.
\item Um método de ordenaçăo é \textbf{estável} se a ordem relativa dos itens com
chaves iguais năo se altera durante a ordenaçăo.
\item Alguns dos métodos de ordenaçăo mais eficientes năo săo estáveis.
\item A estabilidade pode ser forçada quando o método é năo-estável.
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Introduçăo - Conceitos Básicos}
\begin{itemize}
\item Classificaçăo dos métodos de ordenaçăo:
\begin{itemize}
\item Interna: arquivo a ser ordenado cabe todo na memória principal.
\item Externa: arquivo a ser ordenado năo cabe na memória principal.
\end{itemize}
\item Diferenças entre os métodos:
\begin{itemize}
\item Em um método de ordenaçăo interna, qualquer registro pode ser
imediatamente acessado.
\item Em um método de ordenaçăo externa, os registros săo acessados
sequencialmente ou em grandes blocos.
\end{itemize}
\item A maioria dos métodos de ordenaçăo é baseada em \textbf{comparaçőes}
das chaves.
\end{itemize}
\end{frame}


\begin{frame}
\frametitle{Ordenaçăo Interna}
\begin{itemize}
\item  Na escolha de um algoritmo de ordenaçăo interna deve ser
considerado o tempo gasto pela ordenaçăo.
\item  Sendo $n$ o número registros no arquivo, as medidas de complexidade
relevantes săo:
\begin{itemize}
\item Número de comparaçőes $C(n)$ entre chaves.
\item Número de movimentaçőes $M(n)$ de itens do arquivo.
\end{itemize}
\item O uso econômico da memória disponível é um requisito primordial na
ordenaçăo interna.
\item Métodos de ordenaçăo que utilizam a estrutura de um vetor e executam a permutaçăo dos itens no próprio vetor săo os preferidos.
\item Métodos que utilizam listas encadeadas năo săo muito utilizados.
\item Métodos que fazem cópias dos itens a serem ordenados possuem
menor importância
\end{itemize}
\end{frame}


\begin{frame}
\frametitle{Classificaçăo dos métodos de ordenaçăo interna}
\begin{itemize}
\item Métodos simples:
\begin{itemize}
\item Adequados para pequenos arquivos.
\item Requerem $(n^2)$ comparaçőes (onde $n$ é a quantidade de elementos).
\item Produzem programas pequenos.
\end{itemize}
\item Métodos eficientes:
\begin{itemize}
\item Adequados para arquivos maiores.
\item Requerem $(n \times log~n)$ comparaçőes.
\item Usam menos comparaçőes.
\item As comparaçőes săo mais complexas nos detalhes.
\item Métodos simples săo mais eficientes para pequenos arquivos.
\end{itemize}
\end{itemize}

\end{frame}



\begin{frame}
\frametitle{Ordenaçăo Interna}
\begin{itemize}
\item Tipos de dados e variáveis utilizados nos algoritmos de ordenaçăo
interna
\item O índice do vetor vai de $0$ até $MaxTam$, devido ŕs chaves sentinelas
\item O vetor a ser ordenado contém chaves nas posiçőes de $1$ até $n$
\end{itemize}
\end{frame}

\section{Ordenaçăo Bolha}

\begin{frame}
\frametitle{Ordenaçăo Bolha - \textit{Bubble Sort}}
\begin{itemize}
\item O algoritmo de ordenaçăo bolha é simples de entendimento e de fácil implementaçăo. 
\item Está entre os mais conhecidos e difundidos métodos de ordenaçăo de arranjos, mas năo é um algoritmo eficiente.
\item O princípio do bolha é a troca de valores entre posiçőes consecutivas, fazendo com que os valores mais altos (ou mais baixos) "borbulhem" para o final do arranjo (por isso o nome \textit{Bubblesort}). 
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Ordenaçăo Bolha - \textit{Bubble Sort}}
\begin{itemize}
\item A ideia é comparar dois elementos e trocá-los de posiçăo, até que os elementos de maior valor sejam levados para o final do vetor.
\item O processo continua até a ordenaçăo total do vetor lembrando a forma como as bolhas em um tanque de água procuram seu próprio nível.
\item O tempo (custo) computacional desse algoritmo é de ordem quadrática $(n^2)$
\item Tempo computacional: năo se refere ao tempo de relógio e sim ao número de operaçőes (comparaçőes ou trocas) realizadas pelo algoritmo no processo de ordenaçăo em funçăo da quantidade $n$ de itens de entrada. 
\item Ele năo é recomendado para programas que precisem de velocidade e operem com quantidade elevada de dados. 
\end{itemize}
\end{frame}

\section{Ordenaçăo por Seleçăo}



\begin{frame}
\frametitle{Ordenaçăo por Seleçăo - \textit{Selection sort}}
\begin{itemize}
\item Um dos algoritmos mais simples de ordenaçăo.
\item Algoritmo:
\begin{itemize}
\item \textbf{Selecione} o menor item do vetor.
\item Troque-o com o item da primeira posiçăo do vetor.
\item Repita essas duas operaçőes com os $n-1$ itens restantes, depois
com os $n-2$ itens, até que reste apenas um elemento
\end{itemize}
\end{itemize}
\end{frame}

\section{Ordenaçăo por Inserçăo}

\begin{frame}
\frametitle{Ordenaçăo por Inserçăo - \textit{Insertion sort}}
\begin{itemize}
\item Método preferido dos jogadores de cartas.
\begin{itemize}
\item As cartas săo ordenadas da esquerda para direita uma por uma. 
\item O jogador escolhe a segunda carta e verifica se ela deve ficar antes ou na posiçăo que está.
\item Depois a terceira carta é classificada, deslocando-a até sua correta posiçăo
\item O jogador realiza esse procedimento até ordenar todas as cartas
\end{itemize}
\item Alto custo em remover uma carta de uma posiçăo e colocá-la em outra quando a representaçăo é por arranjos
\end{itemize}


\end{frame}

\item Melhor Caso: $n$
\item Pior Caso: $n^2$
\end{itemize}
\item Número Movimentaçőes
\begin{itemize}
\item Melhor Caso: $n$
\item Pior Caso: $n^2$
\end{itemize}

\item Características:
\begin{itemize}
\item O número mínimo de comparaçőes e movimentos ocorre quando os itens estăo originalmente em ordem.
\item O número máximo ocorre quando os itens estăo originalmente na ordem reversa.
\item É o método a ser utilizado quando o arquivo está ``quase'' ordenado.
\item É um bom método quando se deseja adicionar uns poucos itens a um arquivo ordenado, pois o custo é linear.
\item O algoritmo de ordenaçăo por inserçăo é \textbf{estável}.
\end{itemize}
\end{itemize}
\end{frame}
