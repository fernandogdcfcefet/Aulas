void Particao( TipoIndice Esq, TipoIndice Dir , TipoIndice *i , TipoIndice *j , TipoItem *A){
 TipoItem temp, pivo;
 *i = Esq;
 *j = Dir ;
 pivo = A[( *i + *j ) / 2 ] ; 

 do{
  while ( pivo.Chave > A[*i].Chave)
    (*i)++;
  while ( pivo.Chave < A[*j].Chave)
    (*j)--;

  if ( *i <= *j ){
    temp = A[*i] ;
    A[*i] = A[*j] ;
    A[*j] = temp;
    (*i)++;
    (*j)--;
  }
 }while ( *i <= *j );

}
