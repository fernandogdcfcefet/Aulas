\textbf{Resposta}: \\

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
int main(){

    char texto[50][100];
    int indice=0;

    printf("\n*************** Editor de Texto ***************\n");

        printf("Erro ao abrir o arquivo!");
        exit(0);
    }

    for(int i=0; i<indice; i++)
         fprintf(arquivo,"%s\n",texto[i]);

    fclose(arquivo);

    printf("Arquivo salvo com sucesso!\n");

    return 0;
}

\end{lstlisting}

\item Construa um programa que aloque em tempo de  execuçăo (dinamicamente)

#include <stdio.h>
#include <stdlib.h>

int **aloca_matriz(int m, int n){

    int **matriz = (int**)malloc(m*sizeof(int *));

    for(int i=0;i<m;i++)
        matriz[i] = (int*)malloc(n*sizeof(int));

    return matriz;
}

void desaloca_matriz(int **matriz, int m){

    for(int i=0;i<m;i++)
        free(matriz[i]);

    free(matriz);
}

void preenche_matriz(int **matriz, int m, int n){

    for(int i=0;i<m;i++)
        for(int j=0;j<n;j++)
        matriz[i][j] = i+j;

}

void imprimi_matriz(int **matriz, int m, int n){

    for(int i=0;i<m;i++){
        for(int j=0;j<n;j++)
        printf("%d\t",matriz[i][j]);

        printf("\n");
    }
}

int main(){

    int m = 3;
    int n = 5;

    int **nova_matriz = aloca_matriz(m,n);

\section{Análise de Complexidade}
\begin{enumerate}

\item Suponha dois algoritmos $A$ e $B$, com funçőes de complexidade:

\begin{enumerate}
\item $A(n) = n^2 - n + 289$ \\

\textbf{Resposta}:

\begin{equation*}
O(n^2) - O(n) + O(289) = O(n^2)
\end{equation*}

\item $B(n) = 7n + 49$

\textbf{Resposta}:

\begin{equation*}
O(7n) + O(49) = O(n)
\end{equation*}


\end{enumerate}

Determine, utilizando a notaçăo $O$ o grau de complexidade de cada um dos algoritmos.   


\item Determine a complexidade o programa abaixo. O que esse algoritmo faz?
\begin{lstlisting}
int main (){
	int i=0, n=0, s=0, V[100];
	scanf(" %d",&n);
	do{
		scanf("%d",V[i]);
		s = s + V[i++];
	}while(i<n);
	cout<<s/n;
}
\end{lstlisting}

\textbf{Resposta}: \\

O algoritmo calcula a média dos valores recebidos no vetor V.

\begin{equation*}
O(1) + O(1) + O(n) + O(n)*O(1) + O(n)*O(1) + O(1) = O(n);
\end{equation*}



\item O que significa dizer: $O(f(n)) + O(g(n)) = O(max(f(n), g(n)))$ ? \\

\textbf{Resposta}: Esta é um propriedade da notaçăo $O$ que significa que a soma de duas funçőes da notaçăo $O$ ($f(n) e g(n)$) será representada pelo valor máximo entre as duas funçőes. Ou seja, a soma representa o custo máximo para executar as duas funçőes separadamente.

\item Determine a (análise de) complexidade de cada um dos algoritmos abaixo e indique sua ordem de complexidade. Considere o melhor e o pior caso para cada um.

\begin{enumerate}
\item Somar matrizes
\begin{lstlisting}
for(i=0; i<n; i++)
  for(j=0; j<n; j++)
    a[i][j] = b[i][j] + c[i][j];
\end{lstlisting}

\textbf{Resposta}: \\
Melhor Caso: $O(n^2)$ \\
Pior Caso: $O(n^2)$


\item Multiplicar matrizes
\begin{lstlisting}
for(i=0; i<n; i++)
  for(j=0; j<n; j++)
  {
    a[i][j] = 0;
    for(k=0; k<n; k++) 
      a[i][j] += b[i][k] * c[k][j];
  }
\end{lstlisting}

\textbf{Resposta}: \\
Melhor Caso: $O(n^3)$ \\
Pior Caso: $O(n^3)$


\item Encontrar o maior valor de um vetor
\begin{lstlisting}
maior = vet[0];
for(i=1; i<n; i++)
  if(vet[i]>valor)
    maior = vet[i];
\end{lstlisting}

\textbf{Resposta}: \\
Melhor Caso: $O(n)$ \\
Pior Caso: $O(n)$

\item Encontrar o maior e o menor valor de um vetor

\item O máximo divisor comum ($MDC$) de dois números inteiros $x$ e $y$ pode ser calculado 
usando-se uma definiçăo recursiva:  

\begin{equation*}
MDC(x, y) = MDC(x-y, y), ~se ~ x > y
\end{equation*}

Além disso, sabe-se que:

\begin{equation*}
MDC(x, y) = MDC(y, x) 
\end{equation*}


\begin{equation*}
MDC(x, x) = x
\end{equation*}

Exemplo: 
$MDC(10, 6) = MDC(4, 6) = MDC(6, 4) = MDC(2, 4) = MDC(4, 2) = MDC (2,2) = 2$


Crie uma funçăo recursiva para descrever tal definiçăo. Crie, 
também, um algoritmo que leia os dois valores inteiros e utilize a funçăo criada para 
calcular o $MDC$ de $x$ e $y$, e imprima o valor computado.

\textbf{Resposta}: \\

\begin{lstlisting}
int MDC(int x, int y){

    if(x == y)
        return x;
    else
        if(x > y)
            return MDC(x - y, y);
        else return MDC(y - x, x);
}
\end{lstlisting}

    if(y==0)

        soma(predecessor(x),sucessor(y));

void bolha_recursivo(int *A, int n){

    bool trocou=false;

    for(int i=0;i<n-1;i++){
        if(A[i] > A[i+1]){
            int temp = A[i];
            A[i] = A[i+1];
            A[i+1] = temp;
            trocou=true;
        }
    }

    if(trocou) bolha_recursivo(A,n-1);
}

void selecao_recursivo(int *A, int inicio, int n){

    if(inicio < n){

        int min = inicio;

        for(int i=inicio+1; i<n; i++)
            if(A[i] < A[min])
                min = i;

        int temp = A[inicio];
        A[inicio] = A[min];
        A[min] = temp;

        selecao_recursivo(A, inicio+1, n);
    }

}

void inserca_recursivo(int *A, int n){

    if(n > 0){
        inserca_recursivo(A, n-1);
        int x = A[n];
        int i = n-1;
        while(i >=0 && A[i]>x){
            A[i+1] = A[i];
            i--;
        }
        A[i+1]=x;
    }
}
